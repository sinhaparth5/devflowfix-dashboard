{{- else if contains "NodePort" .Values.service.type }}
Get the application URL by running:
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "devflowfix-chart.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
Get the application URL by running:
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "devflowfix-chart.fullname" . }}
{{- else if contains "ClusterIP" .Values.service.type }}
Get the application URL by running:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "devflowfix-chart.fullname" . }} 8080:{{ .Values.service.port }}
  Then visit http://localhost:8080
{{- end }}

Useful commands:
  # Check pod status
  kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "devflowfix-chart.name" . }}

  # View logs
  kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "devflowfix-chart.name" . }} -f

  # Check HPA status
  kubectl get hpa -n {{ .Release.Namespace }} {{ include "devflowfix-chart.fullname" . }}
